        tailwind.config = {

            margin: 0;

        }

    <script src="https://cdnjs.cloudflare.com/ajax/libs/react/18.2.0/umd/react.production.min.js"></script>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/react-dom/18.2.0/umd/react-dom.production.min.js"></script>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/babel-standalone/7.23.5/babel.min.js"></script>

<body>

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useEffect, useRef } = React;


        // LA TUA SCHEDA PERSONALIZZATA

        const MY_WORKOUT = {

          clientName: 'Joseph',

          days: [

            {

              name: 'Giorno 1 - Tirata & Bicipiti',

              exercises: [

                { id: 'g1-ex1', name: 'Rematore', sets: '2', reps: '12', rest: '90' },

                { id: 'g1-ex2', name: 'Lat Machine presa neutra', sets: '2', reps: '6', rest: '120' },

                { id: 'g1-ex3', name: 'Face Pull', sets: '2', reps: '12', rest: '90' },

                { id: 'g1-ex4', name: 'Bicipiti cavi seduto', sets: '2', reps: '12', rest: '90' },

                { id: 'g1-ex5', name: 'Martello', sets: '2', reps: '6', rest: '90' },

                { id: 'g1-ex6', name: 'Addominali', sets: '2', reps: '15', rest: '60' }

              ]

            },

            {

              name: 'Giorno 2 - Spalle, Gambe & Cardio',

              exercises: [

                { id: 'g2-ex1', name: 'Military Press', sets: '2', reps: '12', rest: '120' },

                { id: 'g2-ex2', name: 'Alzate frontali cavo', sets: '2', reps: '12', rest: '90' },

                { id: 'g2-ex3', name: 'Leg Extension', sets: '2', reps: '12', rest: '90' },

                { id: 'g2-ex4', name: 'Pressa', sets: '2', reps: '6', rest: '120' },

                { id: 'g2-ex5', name: 'Adduttori', sets: '2', reps: '12', rest: '90' },

                { id: 'g2-ex6', name: 'Addominali', sets: '2', reps: '15', rest: '60' },

                { id: 'g2-ex7', name: 'Camminata 18% pendenza', sets: '1', reps: '5 min', rest: '0' }

              ]

            },

            {

              name: 'Giorno 3 - Petto & Tricipiti',

              exercises: [

                { id: 'g3-ex1', name: 'Croci ai cavi', sets: '2', reps: '12', rest: '90' },

                { id: 'g3-ex2', name: 'Spinte cavi', sets: '2', reps: '6', rest: '120' },

                { id: 'g3-ex3', name: 'Croci cavi bassi', sets: '2', reps: '12', rest: '90' },

                { id: 'g3-ex4', name: 'Pushdown', sets: '2', reps: '12', rest: '90' },

                { id: 'g3-ex5', name: 'Estensione sopra la testa', sets: '2', reps: '12', rest: '90' }

              ]

            }

          ]

        };


        function App() {

          const [activeDayIndex, setActiveDayIndex] = useState(0);

          const [toastMessage, setToastMessage] = useState(null);

          

          // Memoria locale dei pesi, ripetizioni reali e RPE

          const [gymLogs, setGymLogs] = useState(() => {

            const savedLogs = localStorage.getItem('joseph_stable_logs');

            return savedLogs ? JSON.parse(savedLogs) : {};

          });


          // Stato Timer

          const [timerActive, setTimerActive] = useState(false);

          const [timerDuration, setTimerDuration] = useState(90);

          const [timeLeft, setTimeLeft] = useState(90);

          const timerIntervalRef = useRef(null);


          useEffect(() => {

            localStorage.setItem('joseph_stable_logs', JSON.stringify(gymLogs));

          }, [gymLogs]);


          useEffect(() => {

            if (timerActive && timeLeft > 0) {

              timerIntervalRef.current = setInterval(() => {

                setTimeLeft((prev) => {

                  if (prev <= 1) {

                    clearInterval(timerIntervalRef.current);

                    setTimerActive(false);

                    showToast("⏱️ Recupero terminato! Sotto con la prossima serie!");

                    return 0;

                  }

                  return prev - 1;

                });

              }, 1000);

            } else {

              clearInterval(timerIntervalRef.current);

            }

            return () => clearInterval(timerIntervalRef.current);

          }, [timerActive, timeLeft]);


          const showToast = (msg) => {

            setToastMessage(msg);

            setTimeout(() => setToastMessage(null), 2500);

          };


          const handleLogChange = (exId, setIdx, field, value) => {

            const key = `${exId}-${setIdx}-${field}`;

            setGymLogs(prev => ({ ...prev, [key]: value }));

          };


          const startTimer = (seconds) => {

            setTimerDuration(seconds);

            setTimeLeft(seconds);

            setTimerActive(true);

            showToast(`⏱️ Timer avviato: ${seconds}s`);

          };


          const formatTime = (secs) => {

            const mins = Math.floor(secs / 60);

            const rem = secs % 60;

            return `${mins}:${rem < 10 ? '0' : ''}${rem}`;

          };


          return (

            <div className="min-h-screen bg-slate-900 text-slate-100 font-sans pb-28 select-none">

              {toastMessage && (

                <div className="fixed top-4 left-1/2 -translate-x-1/2 z-50 bg-emerald-500 text-slate-950 font-black px-6 py-3 rounded-full shadow-2xl border border-emerald-400 text-xs uppercase tracking-wider">

                  {toastMessage}

                </div>

              )}


              {/* Header super pulito */}

              <header className="sticky top-0 z-40 bg-slate-900/90 backdrop-blur border-b border-slate-800 px-4 py-3 flex justify-between items-center">

                <div className="flex items-center space-x-2">

                  <span className="bg-emerald-500 text-slate-950 px-2 py-1 rounded-lg font-black text-xs">JF</span>

                  <h1 className="text-sm font-bold tracking-tight">Gym Workout Tracker</h1>

                </div>

                <span className="text-[11px] font-bold text-slate-400 bg-slate-800 px-2 py-1 rounded-md border border-slate-700">Atleta: {MY_WORKOUT.clientName}</span>

              </header>


              <main className="max-w-md mx-auto px-3 mt-4 space-y-4">

                

                {/* Tabs dei tre giorni */}

                <div className="grid grid-cols-3 gap-1 bg-slate-850 p-1 rounded-xl border border-slate-800">

                  {MY_WORKOUT.days.map((day, idx) => (

                    <button

                      key={idx}

                      onClick={() => setActiveDayIndex(idx)}

                      className={`py-2.5 text-[11px] font-black rounded-lg transition-all text-center ${activeDayIndex === idx ? 'bg-slate-700 text-emerald-400 shadow-sm' : 'text-slate-400'}`}

                    >

                      {day.name.split(' - ')[0]}

                    </button>

                  ))}

                </div>


                <h2 className="text-xs font-bold uppercase tracking-widest text-slate-400 px-1">

                  {MY_WORKOUT.days[activeDayIndex]?.name}

                </h2>


                {/* Lista Esercizi del Giorno Selezionato */}

                <div className="space-y-3">

                  {MY_WORKOUT.days[activeDayIndex]?.exercises.map((ex, exIdx) => (

                    <div key={ex.id} className="bg-slate-850 border border-slate-800 rounded-xl p-4 shadow-sm space-y-3">

                      

                      <div className="flex justify-between items-baseline">

                        <h3 className="text-sm font-black text-white">{exIdx + 1}. {ex.name}</h3>

                        <span className="text-[11px] font-bold text-slate-400">Target: {ex.sets}x{ex.reps}</span>

                      </div>


                      {/* Inputs per i Set */}

                      <div className="space-y-2 pt-2 border-t border-slate-800/60">

                        <div className="grid grid-cols-12 text-[9px] uppercase font-black tracking-wider text-slate-500 text-center">

                          <div className="col-span-2 text-left">Set</div>

                          <div className="col-span-4">Kg</div>

                          <div className="col-span-3">Reps</div>

                          <div className="col-span-3">Difficoltà</div>

                        </div>


                        {Array.from({ length: parseInt(ex.sets, 10) || 1 }).map((_, setIdx) => {

                          const weightKey = `${ex.id}-${setIdx}-weight`;

                          const repsKey = `${ex.id}-${setIdx}-reps`;

                          const rpeKey = `${ex.id}-${setIdx}-rpe`;

                          

                          return (

                            <div key={setIdx} className="grid grid-cols-12 items-center gap-1.5 bg-slate-900/40 p-1 rounded-lg">

                              <div className="col-span-2 text-xs font-black text-slate-500 text-center">#{setIdx + 1}</div>

                              

                              {/* Campo KG */}

                              <div className="col-span-4 bg-slate-900 border border-slate-700 rounded-md flex items-center px-1">

                                <input

                                  type="number"

                                  placeholder="-"

                                  value={gymLogs[weightKey] || ''}

                                  onChange={(e) => handleLogChange(ex.id, setIdx, 'weight', e.target.value)}

                                  className="w-full bg-transparent text-center text-xs font-bold py-1.5 focus:outline-none text-emerald-400"

                                />

                                <span className="text-[9px] text-slate-500 font-bold pr-0.5">kg</span>

                              </div>


                              {/* Campo REPS EFFETTIVE */}

                              <div className="col-span-3 bg-slate-900 border border-slate-700 rounded-md">

                                <input

                                  type="number"

                                  placeholder={ex.reps}

                                  value={gymLogs[repsKey] || ''}

                                  onChange={(e) => handleLogChange(ex.id, setIdx, 'reps', e.target.value)}

                                  className="w-full bg-transparent text-center text-xs font-bold py-1.5 focus:outline-none text-blue-400"

                                />

                              </div>


                              {/* Campo RPE */}

                              <div className="col-span-3 bg-slate-900 border border-slate-700 rounded-md">

                                <select

                                  value={gymLogs[rpeKey] || ''}

                                  onChange={(e) => handleLogChange(ex.id, setIdx, 'rpe', e.target.value)}

                                  className="w-full bg-transparent text-center text-xs font-bold py-1.5 focus:outline-none text-amber-400 appearance-none"

                                >

                                  <option value="">RPE</option>

                                  {[10, 9.5, 9, 8.5, 8, 7.5, 7, 6.5, 6].map(v => (

                                    <option key={v} value={v} className="bg-slate-900 text-white">@{v}</option>

                                  ))}

                                </select>

                              </div>

                            </div>

                          );

                        })}

                      </div>

                      

                      {/* Tasto rapido timer dell'esercizio */}

                      {ex.rest !== '0' && (

                        <button 

                          onClick={() => startTimer(parseInt(ex.rest, 10))}

                          className="w-full bg-slate-800 border border-slate-700 text-slate-400 py-1 rounded-md text-[10px] font-bold active:bg-slate-750 transition-all"

                        >

                          ⏳ Serie chiusa! Fai partire {ex.rest}s di pausa

                        </button>

                      )}

                    </div>

                  ))}

                </div>


                {/* Barra di controllo del recupero sempre visibile in basso */}

                <div className="fixed bottom-4 left-3 right-3 max-w-md mx-auto bg-slate-850/95 border border-slate-700 p-3 rounded-xl shadow-2xl flex items-center justify-between backdrop-blur-md z-30">

                  <div>

                    <p className="text-[9px] uppercase tracking-wider font-black text-slate-400">Recupero Attivo</p>

                    <p className="text-xl font-mono font-black text-emerald-400">{formatTime(timeLeft)}</p>

                  </div>

                  <div className="flex space-x-1.5">

                    <button onClick={() => setTimerActive(!timerActive)} className={`px-4 py-2 text-xs font-black rounded-lg ${timerActive ? 'bg-amber-500 text-slate-950' : 'bg-emerald-500 text-slate-950'}`}>

                      {timerActive ? 'Pausa' : 'Avvia'}

                    </button>

                    <button onClick={() => { setTimerActive(false); setTimeLeft(timerDuration); }} className="bg-slate-900 border border-slate-700 px-3 py-2 text-xs font-bold rounded-lg text-slate-400">Reset</button>

                  </div>

                </div>


              </main>

            </div>

          );

        }


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
